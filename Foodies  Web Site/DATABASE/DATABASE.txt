CREATE TABLE CUSTOMER(
CUS_ID INT AUTO_INCREMENT,
CUS_FIRST_NAME VARCHAR(25) NOT NULL,
CUS_MIDDLE_NAME VARCHAR(25) NOT NULL,
CUS_LAST_NAME VARCHAR(25) NOT NULL,
CUS_NUMBER BIGINT UNSIGNED NOT NULL UNIQUE,
CUS_EMAIL VARCHAR(360) NOT NULL,
CUS_ADDRESS VARCHAR(100) NOT NULL,
CUS_PASSWORD VARCHAR(16) NOT NULL,
PRIMARY KEY (CUS_ID)
);

CREATE TABLE RECEIPE(
REC_ID INT AUTO_INCREMENT,
REC_NAME VARCHAR(25) NOT NULL UNIQUE,
REC_DESCRIPTION VARCHAR(100) NOT NULL,
REC_PRICE INT NOT NULL,
REC_PICTURE VARCHAR(100),
PRIMARY KEY (REC_ID)
);

CREATE TABLE ORDERS(
ORD_ID INT AUTO_INCREMENT,
CUS_ID INT NOT NULL,
REC_ID INT NOT NULL,
ORD_DATE DATE NOT NULL,
ORD_TIME TIME NOT NULL,
PRIMARY KEY (ORD_ID)
);
